const retry = useRetryAsync({
    maxRetries: 10,
    pauseDuration: 2000,
    onIterationComplete: (urls, values) => {
        if (values[0].data && values[1].data) {
            retry.reset();
        }
    },
    onComplete: mediaFiles => {
        setLoading(false);
    },

    retry.init([
        () => axios.get("/api/v1/1234/status"),

    ]);
});
